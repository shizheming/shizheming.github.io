                    var obj = formverify(
                        'ok',        //点击提交的id
                        function() {}    //全部验证通过后要干的事情回调
                        
                    );



    ////////////////2.初始完的表单对象拥有2个方法add，remove。
    ////////////////obj.add添加某一项的验证

                    obj.add({
                        name : '辅导员完整的真实姓名',  //给用户的提示信息
                        id : 'instructor',              //控件的id
                        *type : 'tel',                  //内置的验证的选项
                                 tel       //电话  
                                 IDcard    //身份证
                                 mail      //邮箱
                        *regex : /[a-z][0-1]/                     //正则验证
                    });

                    //带*的key可以省略
                    //2个都不写默认验证有没有，2个都写regex权重高于type



    ////////////////obj.remove删除某一项控件验证

                    obj.remove(
                        'instructor'    //控件id
                    );
