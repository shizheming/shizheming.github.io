////倒计时
_.second(100, function() {
	alert(1);
}, function() {
	alert(2);
});
_.second(时间, ****倒计时开始前的运行的函数, ****倒计时结束后运行的函数);


////节流代理（一段时间内只相应一次）
var a = _.proxy(function() {
	console.log(11);
}, 2000);
document.body.addEventListener('click', function() {
	a();
});
_.proxy(要干事情的函数, ****时间 || 1000);


////弹窗
_.popUp('我是弹窗< div>1111', function() {
	console.log(1111);
}, '你好');
_.popUp(弹窗的内容--可以写标签, ****弹窗关闭后干的事情, ****弹窗按钮的文案);


////顶部提示信息
_.title('请稍后再试');
_.title(提示信息的内容);


////判断函数
都返回布尔值
isArguments(val)    是不是伪数组
isFunction(val)    是不是函数
isString(val)    是不是字符串
isNumber(val)    是不是数字
isDate(val)    是不是日期
isRegExp(val)    是不是正则
isArray(val)    是不是数组
